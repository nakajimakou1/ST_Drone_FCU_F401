203 void BSP_PB_Init(Button_TypeDef Button, ButtonMode_TypeDef ButtonMode)
204 {
205  GPIO_InitTypeDef GPIO_InitStruct;
206 
207  /* Enable the BUTTON Clock */
208  BUTTONx_GPIO_CLK_ENABLE(Button);
209  __SYSCFG_CLK_ENABLE();
210 
211  if(ButtonMode == BUTTON_MODE_GPIO)
212  {
213  /* Configure Button pin as input */
214  GPIO_InitStruct.Pin = BUTTON_PIN[Button];
215  GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
216  GPIO_InitStruct.Pull = GPIO_PULLDOWN;
217  GPIO_InitStruct.Speed = GPIO_SPEED_FAST;
218  HAL_GPIO_Init(BUTTON_PORT[Button], &GPIO_InitStruct);
219  }
220 
221  if(ButtonMode == BUTTON_MODE_EXTI)
222  {
223  /* Configure Button pin as input with External interrupt */
224  GPIO_InitStruct.Pin = BUTTON_PIN[Button];
225  GPIO_InitStruct.Pull = GPIO_NOPULL;
226  GPIO_InitStruct.Mode = GPIO_MODE_IT_RISING;
227  HAL_GPIO_Init(BUTTON_PORT[Button], &GPIO_InitStruct);
228 
229  /* Enable and set Button EXTI Interrupt to the lowest priority */
230  HAL_NVIC_SetPriority((IRQn_Type)(BUTTON_IRQn[Button]), 0x0F, 0x00);
231  HAL_NVIC_EnableIRQ((IRQn_Type)(BUTTON_IRQn[Button]));
232  }
233 }
