112 
113  loop->error_add += loop->error_now;
114 
115  if(loop->error_add > loop->error_add_max)
116  { loop->error_add = loop->error_add_max;
117  }//end of if(loop->error_add > loop->error_add_max)

121 
122  loop->error_older = loop->error_old;
123  loop->error_old = loop->error_now;
124

126  ui = loop->KI * loop->error_add;
127  ud = loop->KD * (loop->error_old - loop->error_older);
128  uc = loop->KC * loop->error_saturated;
129 
130  loop->pre_output.all = up + ui + ud + uc;
131
